\newcommand\kanjisheet[1]{%
  \readlist\kanji{#1}%
  
  \begin{tikzpicture}[remember picture, overlay]
    % 
    % 
    \node at ([xshift=1cm, yshift=8.5mm] current page.south west){
      \begin{tikzpicture}[remember picture, overlay]
        \draw[style=grid lines,step=5mm] (0,12cm) grid +(18cm,14cm); 
        \draw[style=dividing lines,step=2cm] (0,0) grid +(18cm,28cm); 
      \end{tikzpicture}
    };
      \foreach \k in {1,...,9}
      {
        \node at ([xshift=2*\k cm, yshift=-2cm] current page.north west) {\stroke\kanji[\k]};
        \node at ([xshift=2*\k cm, yshift=-4cm] current page.north west) {\kanji[\k]};
      }
      \foreach \d in {6,8,10}
      {
        \foreach \k in {1,...,9}
        {
          \node [opacity=0.1] at ([xshift=2*\k cm, yshift=-\d cm] current page.north west) {\kanji[\k]};
        }
      }
  \end{tikzpicture}\clearpage
}
